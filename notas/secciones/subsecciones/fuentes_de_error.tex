\begin{definition}[Redondeo]
    Es el error que puede producir una máquina al realizar un redondeo.
\end{definition}
\begin{defexample}
    \begin{equation*}
        (\sqrt{\pi})^{2} = \pi
    \end{equation*}
\end{defexample}

\begin{definition}[Inherente]
    Es el error que puede percibir un ser humano al realizar una medición.
\end{definition}
\begin{defexample}
    Cuando medimos con una regla, o vemos el velocímetro analógico del auto mientras manejamos, etc.
\end{defexample}

\begin{definition}[Truncamiento]
    Cuando en una discretización o una aproximación por una serie, decidimos despreciar algún término.
\end{definition}
\begin{defexample}
    \begin{equation*}
        \sum_{n=0}^{\infty} \frac{(-1)^{n}}{2n + 1} = \frac{\pi}{4}
    \end{equation*}
\end{defexample}
