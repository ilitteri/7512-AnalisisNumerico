Redondeamos el valor medido $\overline{x}$, y mayoramos la cota de error $\Delta{x}$ (mayorar es reducir el número a un sólo dígito distinto de cero, e.g. $\Delta{x} = 0.0033 < 0.004 \Rightarrow \Delta{x_{mayorado}} = 0.004$).

\begin{equation*}
    x = \overline{x} \pm \Delta x
\end{equation*}
\begin{equation*}
    \Delta x = 0.d_{1} \cdot 10^{-t}
\end{equation*}

\begin{examples*} Convención

    \begin{enumerate}
        \item 
            \begin{equation*}
                \left.
                    \begin{aligned}
                        &\overline{x} = 123.45678\\
                        &\Delta{x} = 0.0033 < 0.004
                    \end{aligned}
                \right\}
                \quad 
                    \begin{aligned}
                        x &= \overline{x} \pm \Delta x\\
                        &= 123.457 \pm 0.004
                    \end{aligned}
            \end{equation*}
        \item 
            \begin{equation*}
                \left.
                    \begin{aligned}
                        &\overline{x} = 123.45678\\
                        &\Delta{x} = 0.0059 < 0.006
                    \end{aligned}
                \right\}
                \quad 
                    \begin{aligned}
                        x &= \overline{x} \pm \Delta x\\
                        &= 123.457 \pm 0.006
                    \end{aligned}
            \end{equation*}
        \item 
            \begin{equation*}
                \left.
                    \begin{aligned}
                        &\overline{x} = 123.12345\\
                        &\Delta{x} = 0.005
                    \end{aligned}
                \right\}
                \quad 
                    \begin{aligned}
                        x &= \overline{x} \pm \Delta x\\
                        &= 123.123 \pm 0.005
                    \end{aligned}
            \end{equation*}
        \item 
            \begin{equation*}
                \left.
                    \begin{aligned}
                        &\overline{x} = 188.141\\
                        &\Delta{x} = 2.18 < 3
                    \end{aligned}
                \right\}
                \quad 
                    \begin{aligned}
                        x &= \overline{x} \pm \Delta x\\
                        &= 188 \pm 3
                    \end{aligned}
            \end{equation*}
        \item 
            \begin{equation*}
                \left.
                    \begin{aligned}
                        &\overline{x} = 211117\\
                        &\Delta{x} = 611 < 700
                    \end{aligned}
                \right\}
                \quad 
                    \begin{aligned}
                        x &= \overline{x} \pm \Delta x\\
                        &= 211100 \pm 700
                    \end{aligned}
            \end{equation*}
    \end{enumerate}
\end{examples*}
